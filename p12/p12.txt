void eraseAllSubStr(string & mainStr, const string & toErase)
{
    size_t pos = std::string::npos;

    while ((pos  = mainStr.find(toErase))!= std::string::npos)
    {
        mainStr.erase(pos, toErase.length());
        if(pos!=0)
            mainStr.insert(pos," ");
    }
}

float power(float x, int y)
{
    float temp;
    if (y == 0)
        return 1;
    temp = power(x, y / 2);
    if (y % 2 == 0)
        return temp * temp;
    else {
        if (y > 0)
            return x * temp * temp;
        else
            return (temp * temp) / x;
    }
}
