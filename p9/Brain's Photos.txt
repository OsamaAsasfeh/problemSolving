void eraseAllSubStr(string & mainStr, const string & toErase)
{
    size_t pos = std::string::npos;
 
    while ((pos  = mainStr.find(toErase))!= std::string::npos)
    {
        mainStr.erase(pos, toErase.length());
        if(pos!=0)
            mainStr.insert(pos," ");
    }
}
